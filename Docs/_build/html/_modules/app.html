def nextKey(data):
    global gdata
    gdata["frameNo"] = data["frameNo"]
    if data["Player"] == None:
        gdata["Danner"] = data["Danner"]
    else:
        gdata["Player"] = data["Player"]
    sio.emit('begin', gdata)
